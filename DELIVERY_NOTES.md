```cpp
#include "eventbus.hpp"
using namespace eventbus;

// 立即开始使用
EventBus bus;
bus.subscribe("event", [](const std::string& msg) {
    std::cout << "Received: " << msg << std::endl;
});
bus.publish("event", "Hello EventBus!");
```
